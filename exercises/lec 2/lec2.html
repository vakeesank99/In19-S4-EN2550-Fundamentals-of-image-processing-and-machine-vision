%matplotlib inline
import numpy as np
import cv2 as cv
import matplotlib.pyplot as plt

f=cv.imread(r'C:\Python39\cv\exercices\lec 2\spider.png', cv.IMREAD_GRAYSCALE)
assert f is not None

gamma = np.array([0.2 ,0.8, 1.2, 2])

for i in gamma:
    t =np.array ([(p/255)**i*255 for p in range(0,256)]).astype(np.uint8)
    g = cv.LUT(f,t)

    fig,ax = plt.subplots()
    ax.plot(t)

    cv.namedWindow('Image',cv.WINDOW_AUTOSIZE)
    cv.imshow('Image', f)
    cv.waitKey(0)
    cv.imshow("Image", g)
    cv.waitKey(0)
    cv.destroyAllWindows()
